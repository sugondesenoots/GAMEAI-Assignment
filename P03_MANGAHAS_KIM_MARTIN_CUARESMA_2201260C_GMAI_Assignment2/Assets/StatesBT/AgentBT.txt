tree("Root") 
    repeat
        fallback
            tree("IdleTree")
            tree("ShoplistState")
            tree("CartState") 
            tree("RetrieveState") 
            tree("ConfirmState") 
            tree("FollowState")  
            tree("PaymentState")  
            tree("PackingState")  
            tree("CollectionState") 
            tree("FeedbackState")

tree("IdleTree")
    sequence
        IsIdleState
        IdleDialogue
        WaitForInteractClick
        SwitchToShoplist

tree("ShoplistState")
    fallback
        sequence
            IsShoplistState 
            AddItemsToCart
            WaitForGoCartClick
            SwitchToCart
        sequence
            IsShoplistState 
            AddItemsToCart
            WaitForBackClick
            BackToIdle

tree("CartState")
    fallback
        sequence
            IsCartState
            WaitForRetrieveClick
            SwitchToRetrieve
        sequence
            IsCartState
            WaitForBackToShoplist
            BackToShoplist 
             
tree("RetrieveState")
    fallback
        sequence
            IsRetrieveState
            RetrieveDialogue
            WalkToShelf
            WalkToPlayerAndConfirm
        sequence
            IsRetrieveState 
            RetrieveDialogue
            WaitForBackClickRS
            BackToCart
             
tree("ConfirmState") 
    fallback 
        sequence  
            IsConfirmState  
            WaitForConfirmClick
            SwitchToFollow 
        sequence
            IsConfirmState  
            WaitForBackClickCS
            BackToRetrieve 
             
tree("FollowState") 
    fallback 
        sequence 
            IsFollowState  
            FollowDialogue
            WaitForFollowClick  
            WalkToCounter
            SwitchToPayment 
             
tree("PaymentState") 
    fallback 
        sequence 
            IsPaymentState 
            PaymentDialogue 
            WaitForPayment
            SwitchToPacking 
             
tree("PackingState") 
    fallback 
        sequence 
            IsPackingState 
            PackingDialogue 
            WaitForPlasticBagClick   
            WalkToBagHolder 
            WalkBackToCounter
            SwitchToCollection 
        sequence 
            IsPackingState 
            PackingDialogue 
            WaitForOwnBagClick  
            WalkToBagHolder 
            WalkBackToCounter
            SwitchToCollection 
             
tree("CollectionState") 
    fallback 
        sequence 
            IsCollectionState 
            CollectionDialogue  
            SpawnObjectOnCounter
            WaitForItemCollection
            SwitchToFeedback
             
tree("FeedbackState") 
    fallback 
        sequence 
            IsFeedbackState 
            FeedbackDialogue 
            Feedback
            SwitchToIdle
